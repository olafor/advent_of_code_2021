  numbers = Array.new
  file = File.open("SonarSweepInput.txt", "r")
  while !file.eof?
    tmp = file.readline.to_i
    numbers.append(tmp)
  end
  
  sum = 0
  prevNumber = numbers.first
  numbers.each do |currentNumber|
    if currentNumber > prevNumber
      sum += 1
    end
    prevNumber = currentNumber
  end
  puts sum
